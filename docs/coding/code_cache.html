 21        return self.connection_pool.pop()

 27    def initialize_database(self):
 28        conn = self._get_connection()
 29        cursor = conn.cursor()
 30        cursor.execute(
 31            """
 32            CREATE TABLE IF NOT EXISTS session (
 33                id INTEGER PRIMARY KEY AUTOINCREMENT,
 34                design_goal TEXT NOT NULL,
 35                parent_session_id INTEGER,
 36                coding_plan TEXT,
 37                created_at TIMESTAMP DEFAULT (STRFTIME('%Y-%m-%d %H:%M:%f', 'NOW')),
 38                FOREIGN KEY (parent_session_id) REFERENCES session(id)
 39            )
 40            """
 41        )
 42
 43        cursor.execute(
 44            """
 45            CREATE TABLE IF NOT EXISTS iteration (
 46                id INTEGER PRIMARY KEY AUTOINCREMENT,
 47                session_id INTEGER NOT NULL,
 48                code TEXT NOT NULL,
 49                feedback TEXT,
 50                is_correct INTEGER DEFAULT 0,  -- 1 for True, 0 for False
 51                is_runnable INTEGER DEFAULT 0, -- 1 for True, 0 for False
 52                created_at TIMESTAMP DEFAULT (STRFTIME('%Y-%m-%d %H:%M:%f', 'NOW')),
 53                FOREIGN KEY (session_id) REFERENCES session(id)
 54            )
 55            """
 56        )
 57
 58        cursor.execute(
 59            """
 60            CREATE TABLE IF NOT EXISTS error (
 61                id INTEGER PRIMARY KEY AUTOINCREMENT,
 62                iteration_id INTEGER NOT NULL,
 63                error_type TEXT CHECK(error_type IN ('syntax', 'runtime')),
 64                error_message TEXT,
 65                error_line INTEGER,
 66                created_at TIMESTAMP DEFAULT (STRFTIME('%Y-%m-%d %H:%M:%f', 'NOW')),
 67                FOREIGN KEY (iteration_id) REFERENCES iteration(id)
 68            )
 69            """
 70        )
 71
 72        conn.commit()
 73        self._return_connection(conn)
 74        logging.info("Database tables initialized.")

 78    def get_session(self, session_id, fields=None):
 79        conn = self._get_connection()
 80        cursor = conn.cursor()
 81        if fields is None:
 82            fields = "*"
 83        else:
 84            fields = ", ".join(fields)
 85        cursor.execute(
 86            f"""
 87            SELECT {fields} FROM session WHERE id = ?
 88            """,
 89            (session_id,),
 90        )
 91        result = cursor.fetchone()
 92        self._return_connection(conn)
 93        if result:
 94            return dict(
 95                zip([description[0] for description in cursor.description], result)
 96            )
 97        else:
 98            logging.warning(f"No session found with ID: {session_id}")
 99            return None

221        self, iteration_id, code=None, feedback=None, is_correct=None, is_runnable=None
222    ):
223        conn = self._get_connection()
224        cursor = conn.cursor()
225        set_clauses = []
226        params = []
227        if code is not None:

235            params.append(int(is_correct))

238            params.append(int(is_runnable))
239        if set_clauses:
240            query = f"UPDATE iteration SET {', '.join(set_clauses)} WHERE id = ?"
241            params.append(iteration_id)
242            cursor.execute(query, tuple(params))
243            conn.commit()
244        self._return_connection(conn)
245        logging.info(f"Iteration with ID {iteration_id} updated.")
